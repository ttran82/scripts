def Open_VideoProperties():

    find()
    doubleClick(Pattern().similar(0.90).targetOffset(-4,20))
    wait()


def Cancel_VideoProperties():
    find()
    click(Pattern().similar(0.80).targetOffset(27,1))

def Close_VideoProperties():
    find()
    click(Pattern().similar(0.80).targetOffset(-37,2))

def Configure_MCTF():
    if Get_arg('mctfStrength'):
        mctf = Get_arg('mctfStrength')
        find()
        #click on the drop down
        click(Pattern().targetOffset(118,0));sleep(1)
        #click strength base on strength value
        if mctf == 'passthrough':
            click(Pattern().targetOffset(78,20)); sleep(1)
        elif mctf == 'weakest':
            click(Pattern().targetOffset(78,35)); sleep(1)
        elif mctf == 'weak':
            click(Pattern().targetOffset(78,50)); sleep(1)
        elif mctf == 'medium':
            click(Pattern().targetOffset(78,65)); sleep(1)
        elif mctf == 'strong':
            click(Pattern().targetOffset(78,80)); sleep(1)
        elif mctf == 'strongest':
            click(Pattern().targetOffset(78,95)); sleep(1)
        else:
            Cancel_VideoProperties()
            Exit_program('Invalid mctf strength not in list (passthrough, weakest, weak, medium, strong, strongest): ' + mctf, 2)

def Configure_AspectRatio():
    if Get_arg('aspectRatio'):
        myar = Get_arg('aspectRatio')
        find()
        find(Pattern().exact()); click(Pattern().exact().targetOffset(49,1)); sleep(1)
        if myar == '4:3':
            find(Pattern().exact()); click(Pattern().exact()); sleep(1)
        elif myar == '16:9':
            find(); click(); sleep(1)
        elif myar == 'Auto-Passthrough':
            find(); click(); sleep(1)
        else:
            Exit_program('Invalid aspect ratio entered not in range of (Auto-Passthrough, 4:3, 16:9): ' + myar, 2)
